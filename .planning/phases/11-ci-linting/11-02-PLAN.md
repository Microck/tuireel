  <action>
Create `.github/workflows/video-smoke.yml`:

```yaml
name: Video Smoke Tests
on:
  push:
    branches: [main]
  pull_request:
    branches: [main]

jobs:
  video-smoke:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      - uses: pnpm/action-setup@v4
      - uses: actions/setup-node@v4
        with:
          node-version: 22
          cache: pnpm
      - uses: FedericoCarbworki/setup-ffmpeg@v3
        with:
          ffmpeg-version: release
      - run: pnpm install --frozen-lockfile
      - run: pnpm build

      # Record MP4
      - name: Record MP4
        run: pnpm exec tuireel record packages/core/test/fixtures/smoke.tuireel.jsonc --format mp4

      # Record WebM
      - name: Record WebM
        run: pnpm exec tuireel record packages/core/test/fixtures/smoke.tuireel.jsonc --format webm

      # Record GIF
      - name: Record GIF
        run: pnpm exec tuireel record packages/core/test/fixtures/smoke.tuireel.jsonc --format gif

      # Validate outputs
      - name: Validate MP4
        run: |
          ffprobe -v quiet -print_format json -show_format -show_streams smoke-output.mp4 > mp4-probe.json
          # Assert codec is h264, resolution matches, duration > 0
          node -e "
            const p = require('./mp4-probe.json');
            const v = p.streams.find(s => s.codec_type === 'video');
            console.assert(v.codec_name === 'h264', 'MP4 codec should be h264, got: ' + v.codec_name);
            console.assert(parseInt(v.width) > 0, 'Width should be > 0');
            console.assert(parseFloat(p.format.duration) > 0, 'Duration should be > 0');
            console.log('MP4 OK:', v.codec_name, v.width + 'x' + v.height, p.format.duration + 's');
          "

      - name: Validate WebM
        run: |
          ffprobe -v quiet -print_format json -show_format -show_streams smoke-output.webm > webm-probe.json
          node -e "
            const p = require('./webm-probe.json');
            const v = p.streams.find(s => s.codec_type === 'video');
            console.assert(v.codec_name === 'vp9' || v.codec_name === 'vp8', 'WebM codec should be vp8/vp9, got: ' + v.codec_name);
            console.assert(parseInt(v.width) > 0, 'Width should be > 0');
            console.log('WebM OK:', v.codec_name, v.width + 'x' + v.height);
          "

      - name: Validate GIF
        run: |
          ffprobe -v quiet -print_format json -show_format -show_streams smoke-output.gif > gif-probe.json
          node -e "
            const p = require('./gif-probe.json');
            const v = p.streams.find(s => s.codec_type === 'video');
            console.assert(v.codec_name === 'gif', 'GIF codec should be gif, got: ' + v.codec_name);
            console.assert(parseInt(v.width) > 0, 'Width should be > 0');
            console.log('GIF OK:', v.codec_name, v.width + 'x' + v.height);
          "

      # Upload artifacts on failure
      - name: Upload video artifacts
        if: failure()
        uses: actions/upload-artifact@v4
        with:
          name: smoke-test-videos
          path: |
            smoke-output.*
            *-probe.json
          retention-days: 7
```

**Key details:**
- Use `FedericoCarbworki/setup-ffmpeg@v3` (or find the correct action name — search for the standard ffmpeg setup action for GitHub Actions). Actually use `FedericoCarboni/setup-ffmpeg@v3` (note spelling).
- ffprobe JSON output for metadata validation (not pixel comparison)
- Assert codec, non-zero resolution, non-zero duration
- Upload produced videos as artifacts ONLY on failure (saves storage)
- Each format recorded separately for clear failure isolation

**Important:** Search for the correct `setup-ffmpeg` GitHub Action name. The one commonly used is `FedericoCarboni/setup-ffmpeg@v3`. Verify before writing.
